int	ft_strcmp(char *s1, char *s2)
{
	int				result;
	int				i;
	int				j;
	unsigned char	c1;
	unsigned char	c2;

	result = 0;
	if (s1 && s2)
	{
		i = 0;
		j = 0;
		while (s1[i] != '\0' || s2[j] != '\0')
		{
				c1 = (unsigned char) s1[i];
				c2 = (unsigned char) s2[j];
				result = c1 - c2;
				if (result != 0)
						return result;
				else
				{
						if (s1[i] != '\0')
								i++;
						if (s2[j] != '\0')
								j++;
				}
		}
	}
	return result;
}

{
	int		result;
	int		i;
	int		j;

	result = 0;
	if (s1 && s2 && n > 0)

		j = 0;
		while ((s1[i] != '\0' || s2[j] != '\0')

			if (s1[i] != '\0')
				i++;
			if (s2[j] != '\0')
				j++;
		}

int	ft_strncmp(char *s1, char *s2, unsigned int n)
{
	int		result;
	int		i;
	int		j;

	result = 0;
	if (s1 && s2 && n > 0)
	{
		i = 0;
		j = 0;
		while ((s1[i] != '\0' || s2[j] != '\0')
			&& ((unsigned int) i < n || (unsigned int) j < n))
		{
			result = (unsigned char) s1[i] - (unsigned char) s2[j];
			if (result != 0)
				return (result);
			if (s1[i] != '\0')
				i++;
			if (s2[j] != '\0')
				j++;
		}
	}
	return (result);
}

char	*ft_strcat(char *dest, char *src)
{
	int		i;
	int		j;

	if (dest && src)
	{
		i = 0;
		while (dest[i] != '\0')
			i++;
		j = 0;
		while (src[j] != '\0')
		{
			dest[i] = src[j];
			i++;
			j++;
		}
		dest[i] = '\0';
	}
	return (dest);
}

char	*ft_strncat(char *dest, char *src, unsigned int nb)
{
	int		i;
	int		j;

	if (dest && src && nb > 0)
	{
		i = 0;
		while (dest[i] != '\0')
			i++;
		j = 0;
		while ((unsigned int) j < nb && src[j] != '\0')
		{
			dest[i] = src[j];
			i++;
			j++;
		}
		dest[i] = '\0';
	}
	return (dest);
}

int	ft_strncmp(char *s1, char *s2, unsigned int n)
{
	int		result;
	int		i;
	int		j;

	result = 0;
	if (s1 && s2 && n > 0)
	{
		i = 0;
		j = 0;
		while ((s1[i] != '\0' || s2[j] != '\0')
			&& ((unsigned int) i < n || (unsigned int) j < n))
		{
			result = (unsigned char) s1[i] - (unsigned char) s2[j];
			if (result != 0)
				return (result);
			if (s1[i] != '\0')
				i++;
			if (s2[j] != '\0')
				j++;
		}
	}
	return (result);
}

char	*ft_strstr(char *str, char *to_find)
{
	int	i;
	int	size;

	if (str && to_find)
	{
		i = 0;

			i++;
		size = i;
		if (size > 0)
		{
			i = 0;
			while (str[i] != '\0' && ft_strncmp(&str[i], to_find, size) != 0)
				i++;
			if (str[i] != '\0')
				return (&str[i]);
		}
	}
	return (NULL);
}

#include <stddef.h>

int	ft_strncmp(char *s1, char *s2, unsigned int n)
{
	int		result;
	int		i;
	int		j;

	result = 0;
	if (s1 && s2 && n > 0)
	{
		i = 0;
		j = 0;
		while ((s1[i] != '\0' || s2[j] != '\0')
			&& ((unsigned int) i < n || (unsigned int) j < n))
		{
			result = (unsigned char) s1[i] - (unsigned char) s2[j];
			if (result != 0)
				return (result);
			if (s1[i] != '\0')
				i++;
			if (s2[j] != '\0')
				j++;
		}
	}
	return (result);
}

char	*ft_strstr(char *str, char *to_find)
{
	int	i;
	int	size;

	if (str && to_find)
	{
		i = 0;
		while (to_find[i] != '\0')
			i++;
		size = i;
		if (size == 0)
			return (str);
		else
		{
			i = 0;
			while (str[i] != '\0' && ft_strncmp(&str[i], to_find, size) != 0)
				i++;
			if (str[i] != '\0')
				return (&str[i]);
		}
	}
	return (NULL);
}

unsigned int	ft_strlcat(char *dest, char *src, unsigned int size)
{
	int		i;
	int		j;
	unsigned int	dest_len;

	if (dest && src && size > 0)
	{
		i = 0;
		while (dest[i] != '\0')
			i++;
		dest_len = (unsigned int) i;
		j = 0;
		while ((unsigned int) j + dest_len < size && src[j] != '\0')
		{
			dest[i] = src[j];
			i++;
			j++;
		}
		if (i <= size)
			dest[i] = '\0';
		return((unsigned int) i);
	}
	return (0);
}

#include <stddef.h>

int	ft_strncmp(char *s1, char *s2, unsigned int n)
{
	int		result;
	int		i;
	int		j;

	result = 0;
	if (s1 && s2 && n > 0)
	{
		i = 0;
		j = 0;
		while ((s1[i] != '\0' || s2[j] != '\0')
			&& ((unsigned int) i < n || (unsigned int) j < n))
		{
			result = (unsigned char) s1[i] - (unsigned char) s2[j];
			if (result != 0)
				return (result);
			if (s1[i] != '\0')
				i++;
			if (s2[j] != '\0')
				j++;
		}
	}
	return (result);
}

char	*ft_strstr(char *str, char *to_find)
{
	int	i;
	int	size;

	if (str && to_find)
	{
		i = 0;
		while (to_find[i] != '\0')
			i++;
		size = i;
		if (size == 0)
			return (str);
		else
		{
			i = 0;
			while (str[i] != '\0' && ft_strncmp(&str[i], to_find, size) != 0)
				i++;
			if (str[i] != '\0')
				return (&str[i]);
		}
	}
	return (NULL);
}
